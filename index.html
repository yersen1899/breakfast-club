    <div class="card">

        <h2>🛒 Mitbring-Liste</h2>

        <div class="row">

            <input type="text" id="itemName" placeholder="Was? (z.B. Brötchen)">

            <input type="text" id="userName" placeholder="Dein Name">

            <button onclick="addItem()">Hinzufügen</button>

        </div>

        <div id="itemList"></div>

    <div class="card">

        <h2>💬 Talk-Themen</h2>

        <div class="row">

            <input type="text" id="topicInput" placeholder="Worüber reden?">

            <select id="topicType">

                <option value="good">👍 Ja, bitte!</option>

                <option value="bad">🚫 Lieber nicht</option>

            </select>

            <button onclick="addTopic()">Posten</button>

    <script>

        // Aktuell lokale Speicherung (nur auf deinem Gerät sichtbar)

        function render(id, storageKey, templateFn) {

            const data = JSON.parse(localStorage.getItem(storageKey) || '[]');

            const container = document.getElementById(id);

            container.innerHTML = data.length === 0 ? '<p style="color:#999; font-size:0.9rem;">Noch keine Einträge...</p>' : data.map(templateFn).join('');

        }


        function addDate() {

            const val = document.getElementById('dateInput').value;

            if(!val) return;

            const data = JSON.parse(localStorage.getItem('dates') || '[]');

            data.push({date: val});

            localStorage.setItem('dates', JSON.stringify(data));

            loadAll();

            document.getElementById('dateInput').value = '';

        }


        function addItem() {

            const name = document.getElementById('itemName').value;

            const qty = document.getElementById('itemQty').value || 1;

            const user = document.getElementById('userName').value;

            if(!name || !user) return alert("Bitte was und wer eintragen!");

            const data = JSON.parse(localStorage.getItem('items') || '[]');

            data.push({name, qty, user});

            localStorage.setItem('items', JSON.stringify(data));

            loadAll();

            document.getElementById('itemName').value = '';

            document.getElementById('itemQty').value = '';

        }


        function addTopic() {

            const text = document.getElementById('topicInput').value;

            const type = document.getElementById('topicType').value;

            if(!text) return;

            const data = JSON.parse(localStorage.getItem('topics') || '[]');

            data.push({text, type});

            localStorage.setItem('topics', JSON.stringify(data));

            loadAll();

            document.getElementById('topicInput').value = '';

        }


        function loadAll() {

            render('dateList', 'dates', d => `<div class="list-item"><span>${new Date(d.date).toLocaleDateString('de-DE', {weekday: 'long', day: '2-digit', month: '2-digit'})}</span></div>`);

            render('itemList', 'items', i => `<div class="list-item"><span><b>${i.qty}x</b> ${i.name}</span> <span class="tag">von ${i.user}</span></div>`);

            render('topicList', 'topics', t => `<div class="list-item ${t.type}"><span>${t.text}</span></div>`);

        }


        window.onload = loadAll;

    </script>
